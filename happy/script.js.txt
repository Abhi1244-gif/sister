// Home page love message
function showLoveMessage() {
    const message = document.getElementById('love-message');

    message.classList.remove('hidden');

    createConfetti();
    
    setTimeout(() => {
        message.innerHTML += '<br><br>🌟 Made with love just for you! 🌟';

// Flower interactions
let flowerCount = 0;

document.addEventListener('DOMContentLoaded', function() {
    // Flower page click to plant
    const flowersContainer = document.getElementById('flowers');
    if (flowersContainer) {
        flowersContainer.addEventListener('click', plantFlower);
    }
    
    // Auto rainbow rain on flowers page
    if (document.querySelector('.flower-container')) {
        setTimeout(rainbowRain, 3000);
    }
});

function plantFlower(e) {
    const flowersContainer = document.getElementById('flowers');
    const flower = document.createElement('div');
    flower.className = 'flower';
    flower.innerHTML = getRandomFlower();
    
    const rect = flowersContainer.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    flower.style.left = Math.min(x - 25, rect.width - 50) + 'px';
    flower.style.top = Math.min(y - 25, rect.height - 50) + 'px';
    
    flowersContainer.appendChild(flower);
    flowerCount++;
    
    // Remove old flowers to prevent lag (keep max 30)
    if (flowerCount > 30) {
        const oldFlower = flowersContainer.querySelector('.flower');
        if (oldFlower) {
            oldFlower.remove();
            flowerCount--;
        }
    }
    
    // Flower click animation
    flower.addEventListener('click', function() {
        this.style.transform = 'scale(1.5) rotate(360deg)';
        setTimeout(() => {
            this.style.transform = 'scale(1)';
        }, 500);
    });
}

function getRandomFlower() {
    const flowers = ['🌸', '🌺', '🌻', '🌹', '🌷', '🪷', '💐', '🌼', '🌺', '🌸'];
    return flowers[Math.floor(Math.random() * flowers.length)];
}

function blowWind() {
    const flowers = document.querySelectorAll('.flower');
    flowers.forEach((flower, index) => {
        setTimeout(() => {
            flower.style.transition = 'transform 2s ease-out';
            flower.style.transform = `translateX(${Math.random() * 200 - 100}px) translateY(-50px)`;
        }, index * 100);
    });
}

function rainbowRain() {
    for (let i = 0; i < 30; i++) {
        setTimeout(() => {
            const drop = document.createElement('div');
            drop.innerHTML = getRandomSparkle();
            drop.style.cssText = `
                position: fixed;
                left: ${Math.random() * 100}vw;
                top: -20px;
                font-size: 1.5rem;
                pointer-events: none;
                z-index: 1000;
                animation: rainbowFall 4s linear forwards;
            `;
            document.body.appendChild(drop);
            
            setTimeout(() => {
                if (drop.parentNode) drop.remove();
            }, 4000);
        }, i * 150);
    }
}

function getRandomSparkle() {
    const sparkles = ['✨', '🌟', '💫', '⭐', '🌈', '💖'];
    return sparkles[Math.floor(Math.random() * sparkles.length)];
}

// Memory slideshow
let currentSlide = 0;
const slides = document.querySelectorAll('.memory-slide');

function showSlide(index) {
    if (slides.length === 0) return;
    
    slides.forEach(slide => slide.classList.remove('active'));
    slides[index].classList.add('active');
}

function nextSlide() {
    currentSlide = (currentSlide + 1) % slides.length;
    showSlide(currentSlide);
}

function prevSlide() {
    currentSlide = (currentSlide - 1 + slides.length) % slides.length;
    showSlide(currentSlide);
}

// Auto slideshow for memories page
document.addEventListener('DOMContentLoaded', function() {
    if (slides.length > 0) {
        showSlide(currentSlide);
        setInterval(nextSlide, 4000);
        
        // Keyboard navigation
        document.addEventListener('keydown', function(e) {
            if (e.key === 'ArrowRight') nextSlide();
            if (e.key === 'ArrowLeft') prevSlide();
        });
    }
});

// Confetti effect for home page
function createConfetti() {
    for (let i = 0; i < 100; i++) {
        setTimeout(() => {
            const confetti = document.createElement('div');
            confetti.innerHTML = getRandomConfetti();
            confetti.style.cssText = `
                position: fixed;
                left: ${Math.random() * 100}vw;
                top: -10px;
                font-size: 1rem;
                pointer-events: none;
                z-index: 1001;
                animation: confettiFall ${2 + Math.random() * 3}s linear forwards;
            `;
            document.body.appendChild(confetti);
            
            setTimeout(() => {
                if (confetti.parentNode) confetti.remove();
            }, 5000);
        }, i * 20);
    }
}

function getRandomConfetti() {
    const confettiTypes = ['🎉', '🎊', '💖', '🌸', '⭐', '💫', '✨'];
    return confettiTypes[Math.floor(Math.random() * confettiTypes.length)];
}

// Mouse trail effect
document.addEventListener('mousemove', function(e) {
    if (Math.random() < 0.1) { // 10% chance
        const sparkle = document.createElement('div');
        sparkle.innerHTML = '✨';
        sparkle.style.cssText = `

            left: ${e.clientX}px;
            top: ${e.clientY}px;
            font-size: 1rem;
            pointer-events: none;
            z-index: 1002;
            animation: sparkleFade 1s ease-out forwards;
        `;
        document.body.appendChild(sparkle);
        setTimeout(() => sparkle.remove(), 1000);
    }
});

// Navbar smooth scroll effect
document.querySelectorAll('a[href]').forEach(anchor => {
    anchor.addEventListener('click', function (e) {
        e.preventDefault();
        const target = document.querySelector(this.getAttribute('href'));
        if (target) {
            target.scrollIntoView({ behavior: 'smooth' });
        }
    });
});
